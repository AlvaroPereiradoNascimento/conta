#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main (){
	float x = 2, y = 3,c=1;
	printf("1 +");
	while(x <= 50){
		printf( " %.0f/%.0f +", x,y);
		c += x/y;
		x++;
		y+=2;
	}
	printf(" =  %.2f", c);
}
