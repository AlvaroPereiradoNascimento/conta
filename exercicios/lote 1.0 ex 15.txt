#include <stdio.h>

int main (){

   	unsigned int x,y, m;

	scanf("%i", & x);

	scanf("%i", & y);

	x=pow(x,2);
	y=pow (y,2);
	m=sqrt((x+y));

	printf (" A hipotenusa e: %d \n\n", m);

}
